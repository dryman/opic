85 // Simple fast universal hasing
86 // possible way to avoid collision: linear probing + simple tabular hasing
87 // size_t idx = ((size_t)ISA * 31) >> (sizeof(size_t)*8 - 16);
88  /*
89  In future we should use debug flag to enable these
90  printf("ISA matched. ISA: %p, fn: %p\n", method.isa, method.fn); \
91  printf("ISA mismatch.\n"); \
92  */
93 #define OP_TYPECLASS_METHOD_FACTORY(OP_TYPE, METHOD, ISA,...) \
94  do { \
95  Class* isa = (Class*)(((size_t)(ISA)) & (size_t)(~(0x0FL))); \
96  static AtomicClassMethod method_cache[16]; \
97  size_t idx = ((size_t)isa >> 3) & 0x0F; \
98  op_assert(isa, "Class ISA is null\n"); \
99  ClassMethod method; \
100  method = atomic_load(&method_cache[idx]); \
101  OP_METHOD_TYPE(METHOD)* fn = NULL; \
102  if (method.isa == isa) { \
103  fn = (OP_METHOD_TYPE(METHOD)*) method.fn; \
104  } else { \
105  TypeClass** trait_it = isa->traits; \
106  int i=0; \
107  for (TypeClass** trait_it = isa->traits; *trait_it; trait_it++) {\
108  if(!strcmp((*trait_it)->name, #OP_TYPE)) { \
109  OP_TYPE* tc = *(OP_TYPE**) trait_it; \
110  fn = tc->METHOD; \
111  break; \
112  } \
113  } \
114  op_assert(fn,"Class %s does implement %s.%s\n", isa->classname,#OP_TYPE,#METHOD); \
115  method = (ClassMethod){.isa = isa, .fn = (void*) fn}; \
116  atomic_store(&method_cache[idx], method); \
117  } \
118  return fn(__VA_ARGS__); \
119  } while (0);
