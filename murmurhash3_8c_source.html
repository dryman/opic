95 void MurmurHash3_x64_128 ( const void * key, const int len,
96  const uint32_t seed, void * out )
97 {
98  const uint8_t * data = (const uint8_t*)key;
99  const int nblocks = len / 16;
100 
101  uint64_t h1 = seed;
102  uint64_t h2 = seed;
103 
104  const uint64_t c1 = BIG_CONSTANT(0x87c37b91114253d5);
105  const uint64_t c2 = BIG_CONSTANT(0x4cf5ad432745937f);
106 
107  //----------
108  // body
109 
110  const uint64_t * blocks = (const uint64_t *)(data);
111 
112  for(int i = 0; i < nblocks; i++)
113  {
114  uint64_t k1 = getblock64(blocks,i*2+0);
115  uint64_t k2 = getblock64(blocks,i*2+1);
116 
117  k1 *= c1; k1 = ROTL64(k1,31); k1 *= c2; h1 ^= k1;
118 
119  h1 = ROTL64(h1,27); h1 += h2; h1 = h1*5+0x52dce729;
120 
121  k2 *= c2; k2 = ROTL64(k2,33); k2 *= c1; h2 ^= k2;
122 
123  h2 = ROTL64(h2,31); h2 += h1; h2 = h2*5+0x38495ab5;
124  }
125 
126  //----------
127  // tail
128 
129  const uint8_t * tail = (const uint8_t*)(data + nblocks*16);
130 
131  uint64_t k1 = 0;
132  uint64_t k2 = 0;
133 
134  switch(len & 15)
135  {
136  case 15: k2 ^= ((uint64_t)tail[14]) << 48;
137  case 14: k2 ^= ((uint64_t)tail[13]) << 40;
138  case 13: k2 ^= ((uint64_t)tail[12]) << 32;
139  case 12: k2 ^= ((uint64_t)tail[11]) << 24;
140  case 11: k2 ^= ((uint64_t)tail[10]) << 16;
141  case 10: k2 ^= ((uint64_t)tail[ 9]) << 8;
142  case 9: k2 ^= ((uint64_t)tail[ 8]) << 0;
143  k2 *= c2; k2 = ROTL64(k2,33); k2 *= c1; h2 ^= k2;
144 
145  case 8: k1 ^= ((uint64_t)tail[ 7]) << 56;
146  case 7: k1 ^= ((uint64_t)tail[ 6]) << 48;
147  case 6: k1 ^= ((uint64_t)tail[ 5]) << 40;
148  case 5: k1 ^= ((uint64_t)tail[ 4]) << 32;
149  case 4: k1 ^= ((uint64_t)tail[ 3]) << 24;
150  case 3: k1 ^= ((uint64_t)tail[ 2]) << 16;
151  case 2: k1 ^= ((uint64_t)tail[ 1]) << 8;
152  case 1: k1 ^= ((uint64_t)tail[ 0]) << 0;
153  k1 *= c1; k1 = ROTL64(k1,31); k1 *= c2; h1 ^= k1;
154  };
155 
156  //----------
157  // finalization
158 
159  h1 ^= len; h2 ^= len;
160 
161  h1 += h2;
162  h2 += h1;
163 
164  h1 = fmix64(h1);
165  h2 = fmix64(h2);
166 
167  h1 += h2;
168  h2 += h1;
169 
170  ((uint64_t*)out)[0] = h1;
171  ((uint64_t*)out)[1] = h2;
172 }
