103 static void
104 test_BasicInsert(void** context)
105 {
106  OPHeap* heap;
107  RobinHoodHash* rhh;
108 
109  OP_LOG_INFO(logger, "Starting basic insert");
110  assert_true(OPHeapNew(&heap));
111  assert_true(RHHNew(heap, &rhh, 20,
112  0.80, sizeof(int), 0));
113  OP_LOG_DEBUG(logger, "RHH addr %p", rhh);
114  for (int i = 0; i < TEST_OBJECTS; i++)
115  {
116  RHHInsert(rhh, &i, NULL);
117  }
118  RHHPrintStat(rhh);
119  assert_int_equal(TEST_OBJECTS, RHHObjcnt(rhh));
120  ResetObjcnt();
121  RHHIterate(rhh, CountObjects, NULL);
122  assert_int_equal(TEST_OBJECTS, objcnt);
123  ResetObjmap();
124  RHHIterate(rhh, CheckObjects, NULL);
125  for (int i = 0; i < TEST_OBJECTS; i++)
126  {
127  assert_int_equal(1, objmap[i]);
128  }
129  RHHDestroy(rhh);
130  OPHeapDestroy(heap);
131 }
